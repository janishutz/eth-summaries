\subsection{Stetige Verteilungen}

\subsubsection{Gleichverteilung}

\shortdefinition $\cX \sim \cU([a, b])$, falls
$f_\cX = \begin{cases}
        \frac{1}{b - a} & x \in [a, b] \\
        0               & \text{sonst}
    \end{cases}$

\shortremark $\P[\cX \in [c, c + l]] = \frac{l}{b - a}$,
$F_\cX(x) = \begin{cases}
        0                   & x < a           \\
        \frac{x - a}{b - a} & a \leq x \leq b \\
        1                   & x > b
    \end{cases}$
